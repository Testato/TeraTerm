include

インクルードファイルに移る。 

include <include file name>

例

-----メインファイル"main.ttl"------
i=10
:loop
; インクルードファイルへ移る。
include 'sub.ttl'
if i>=0 goto loop
end
-----メインファイル終わり----------

-----インクルードファイル"sub.ttl"----
if i<0 then
  messagebox 'error!' 'sub'
  ; メインファイルへ戻る。
  exit
endif
i = i - 1
; メインファイルへ戻る。
-----インクルードファイル終わり-------
